select "Растения".id_растения, "Растения".id_типа, "Растения".id_семейства, "Растения".название, "Растения".стоимость, "Растения".место_происхождения from "Растения" join "Типы_растений" on "Растения".id_типа = "Типы_растений".id_типа where место_происхождения='Россия';

SELECT растения1.название, растения1.стоимость
FROM "Растения" растения1
WHERE NOT EXISTS (
    SELECT 1
    FROM (
        SELECT DISTINCT особенность
        FROM "Характеристики"
    ) харка
    WHERE NOT EXISTS (
        SELECT 1
        FROM "Характеристики" харка2
        WHERE харка2.id_растения = растения1.id_растения
        AND харка2.особенность = харка.особенность
    )
);
